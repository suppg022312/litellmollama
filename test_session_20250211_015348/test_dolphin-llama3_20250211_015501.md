import pygame

# Initialize Pygame
pygame.init()

# Set up some constants
WIDTH, HEIGHT = 640, 480
BACKGROUND_COLOR = (0, 0, 0)
SHIP_SPEED = 3
ASTERIOD_SPEED = 2

# Load assets
ship_img = pygame.image.load('path_to_your_ship_image.png')
asteroid_img = pygame.image.load('path_to_your_asteroid_image.png')

class Ship:
    def __init__(self):
        self.x = WIDTH / 2
        self.y = HEIGHT / 2
        self.speed = SHIP_SPEED
        self.image = ship_img

    def move(self):
        self.x += self.speed

    def draw(self, screen):
        screen.blit(self.image, (self.x, self.y))

class Asteroid:
    def __init__(self, x, y):
        self.x = x
        self.y = y
        self.speed = ASTEROID_SPEED
        self.image = asteroid_img

    def move(self):
        self.x -= self.speed

    def draw(self, screen):
        screen.blit(self.image, (self.x, self.y))

# Create the ship and asteroids
ship = Ship()
asteroids = [Asteroid(WIDTH - 50, random.randint(100, HEIGHT - 100)) for _ in range(10)]

# Set up the game window
screen = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption('Asteroid Game')
screen.fill(BACKGROUND_COLOR)

# Main game loop
running = True
while running:
    # Handle events (e.g. quitting the game)
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    # Update game objects
    ship.move()
    for asteroid in asteroids:
        asteroid.move()

    # Fill the background with the specified color
    screen.fill(BACKGROUND_COLOR)

    # Draw game objects
    ship.draw(screen)
    for asteroid in asteroids:
        asteroid.draw(screen)

    # Update the display
    pygame.display.flip()

# Clean up and quit the game
pygame.quit()
