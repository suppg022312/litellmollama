import pygame

# Initialize Pygame
pygame.init()

# Set up some constants
SCREEN_WIDTH = 800
SCREEN_HEIGHT = 600
BALL_RADIUS = 10
PADDLE_WIDTH = 100
PADDLE_HEIGHT = 20
FPS = 30

# Set up the display
screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT))
pygame.display.set_caption('Asteroid Game')

# Set up the player paddle
paddle_rect = pygame.Rect(SCREEN_WIDTH / 2 - PADDLE_WIDTH / 2,
                          SCREEN_HEIGHT - PADDLE_HEIGHT,
                          PADDLE_WIDTH,
                          PADDLE_HEIGHT)

# Set up the ball
ball_rect = pygame.Rect(SCREEN_WIDTH / 2, SCREEN_HEIGHT / 2, BALL_RADIUS * 2, BALL_RADIUS * 2)
ball_speed_x = 5
ball_speed_y = -5

# Game loop
running = True
while running:
    # Event handling
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False
            
    # Player paddle movement
    keys = pygame.key.get_pressed()
    if keys[pygame.K_LEFT]:
        paddle_rect.x -= 5
    elif keys[pygame.K_RIGHT]:
        paddle_rect.x += 5
    
    # Ball movement
    ball_rect.x += ball_speed_x
    ball_rect.y += ball_speed_y

    # Bounce the ball off the walls
    if ball_rect.left <= 0 or ball_rect.right >= SCREEN_WIDTH:
        ball_speed_x *= -1
        
    if ball_rect.top <= 0 or ball_rect.bottom >= SCREEN_HEIGHT:
        ball_speed_y *= -1

    # Bounce the ball off the paddle
    if paddle_rect.colliderect(ball_rect):
        ball_speed_y *= -1

    # Draw everything
    screen.fill((0, 0, 0))
    pygame.draw.rect(screen, (255, 255, 255), paddle_rect)
    pygame.draw.ellipse(screen, (255, 255, 255), ball_rect)
    pygame.display.flip()

    # Cap the frame rate
    pygame.time.Clock().tick(FPS)

# Quit Pygame
pygame.quit()
