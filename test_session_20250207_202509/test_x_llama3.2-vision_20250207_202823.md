import pygame
import math
import random

# Window dimensions
WIDTH, HEIGHT = 640, 480

# Colors
WHITE = (255, 255, 255)
BLACK = (0, 0, 0)

class Asteroid:
    def __init__(self):
        self.x = random.randint(0, WIDTH)
        self.y = random.randint(0, HEIGHT)
        self.size = random.randint(10, 50)
        self.speed_x = random.uniform(-2, 2)
        self.speed_y = random.uniform(-2, 2)

    def move(self):
        self.x += self.speed_x
        self.y += self.speed_y

        if self.x < 0 or self.x > WIDTH:
            self.speed_x *= -1
        if self.y < 0 or self.y > HEIGHT:
            self.speed_y *= -1

class Ship:
    def __init__(self):
        self.x = WIDTH / 2
        self.y = HEIGHT / 2
        self.speed_x = 0
        self.speed_y = 0
        self.angle = 0

    def move(self):
        self.x += self.speed_x
        self.y += self.speed_y

class Bullet:
    def __init__(self, x, y, angle):
        self.x = x
        self.y = y
        self.speed_x = math.cos(angle) * 5
        self.speed_y = -math.sin(angle) * 5

def main():
    pygame.init()
    screen = pygame.display.set_mode((WIDTH, HEIGHT))
    clock = pygame.time.Clock()

    ship = Ship()
    asteroids = [Asteroid() for _ in range(10)]
    bullets = []

    running = True
    while running:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                running = False

        keys = pygame.key.get_pressed()
        if keys[pygame.K_LEFT]:
            ship.angle += 5
        if keys[pygame.K_RIGHT]:
            ship.angle -= 5
        if keys[pygame.K_UP]:
            ship.speed_x += math.cos(ship.angle) * 0.1
            ship.speed_y -= math.sin(ship.angle) * 0.1

        if keys[pygame.K_SPACE]:
            bullets.append(Bullet(ship.x, ship.y, ship.angle))

        screen.fill(BLACK)

        for asteroid in asteroids:
            asteroid.move()
            pygame.draw.circle(screen, WHITE, (int(asteroid.x), int(asteroid.y)), asteroid.size)

        for bullet in bullets:
            bullet.move()
            pygame.draw.line(screen, WHITE, (bullet.x, bullet.y), (bullet.x + 10 * math.cos(bullet.speed_x), bullet.y - 10 * math.sin(bullet.speed_y)))
            if bullet.x < 0 or bullet.x > WIDTH or bullet.y < 0 or bullet.y > HEIGHT:
                bullets.remove(bullet)

        ship.move()
        pygame.draw.line(screen, WHITE, (ship.x, ship.y), (ship.x + 20 * math.cos(ship.angle), ship.y - 20 * math.sin(ship.angle)))

        pygame.display.flip()
        clock.tick(60)

    pygame.quit()

if __name__ == "__main__":
    main()
