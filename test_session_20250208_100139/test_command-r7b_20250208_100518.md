def print_board(board):
    """Prints the current state of the board."""
    print("-------------")
    for row in board:
        print("|", end="")
        for cell in row:
            print(f" {cell} |", end="")
        print("\n-------------")

def check_win(board, player):
    """Checks if the given player has won."""
    # Check rows
    for row in board:
        if all([cell == player for cell in row]):
            return True

    # Check columns
    for col in range(3):
        if all([board[row][col] == player for row in range(3)]):
            return True

    # Check diagonals
    if all([board[i][i] == player for i in range(3)]) or all([board[i][2-i] == player for i in range(3)]):
        return True

    return False

def is_board_full(board):
    """Checks if the board is full."""
    return all([cell != ' ' for row in board for cell in row])

def get_player_move(player):
    """Gets the player's move (1 or 2) from the user."""
    while True:
        try:
            move = int(input(f"Player {player}, enter your move (1 or 2): "))
            if move not in [1, 2]:
                print("Invalid move. Please choose 1 or 2.")
            else:
                return move
        except ValueError:
            print("Please enter a valid number.")

def main():
    board = [[' ' for _ in range(3)] for _ in range(3)]
    current_player = 'X'

    while True:
        print_board(board)

        try:
            move = get_player_move(current_player) - 1
            row, col = divmod(move, 3)

            if board[row][col] == ' ':
                board[row][col] = current_player

                if check_win(board, current_player):
                    print_board(board)
                    print(f"Player {current_player} wins!")
                    break
                elif is_board_full(board):
                    print_board(board)
                    print("It's a tie!")
                    break

                current_player = 'O' if current_player == 'X' else 'X'
            else:
                print("That cell is already occupied. Try again.")
        except IndexError:
            print("Invalid move. Board coordinates must be between 1 and 9.")

if __name__ == "__main__":
    main()
